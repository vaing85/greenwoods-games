### AI Gaming Assistant
```javascript
// Get betting recommendation
const recommendation = await fetch('/api/phase8/ai/recommendation', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ 
    userId: 'user123', 
    currentGame: 'blackjack', 
    currentBalance: 1000 
  })
});

// Track user behavior
await fetch('/api/phase8/ai/track-behavior', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ 
    userId: 'user123',

  })
});
```

### Tournament System
```javascript
// Create tournament
const tournament = await fetch('/api/phase8/tournaments', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    name: 'Friday Night Poker',
    game: 'poker',
    type: 'elimination',
    maxPlayers: 16,
    prizePool: [{ position: 1, amount: 1000 }, { position: 2, amount: 500 }]
  })

// Register for tournament
await fetch(`/api/phase8/tournaments/${tournamentId}/register`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    playerId: 'user123',
    playerData: { name: 'Player Name', level: 'Intermediate' }
  })
});
```
